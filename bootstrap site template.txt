        const pre = document.getElementById('preloader');
        
        // Ensure eyes animate slower if you change this later
        document.documentElement.style.setProperty('--loader-speed', '12s');

        // Keep preloader for extra 3 seconds
        setTimeout(() => {
            pre.classList.add('hidden');
            pre.addEventListener('transitionend', () => pre.remove());
        }, 2000);

        const backBtn = document.querySelector('.back-to-top');

        window.addEventListener('scroll', () => {
            if (window.scrollY > 200) {
            backBtn.classList.add('visible');
            } else {
            backBtn.classList.remove('visible');
            }
        });

        backBtn.addEventListener('click', () => {
            window.scrollTo({ top: 0, behavior: 'smooth' });
        });
    </script>
    
    <script>
        const dot = document.querySelector('.cursor-dot');

        let mouseX = 0;
        let mouseY = 0;
        let dotX = 0;
        let dotY = 0;

        // Update mouse position on move
        window.addEventListener('mousemove', (e) => {
            mouseX = e.clientX;
            mouseY = e.clientY;
        });

        // Smooth animation loop using lerp
        function animateDot() {
            // Interpolation speed (0.1 = slow, 1 = instant)
            dotX += (mouseX - dotX) * 0.15;
            dotY += (mouseY - dotY) * 0.15;

            dot.style.left = `${dotX}px`;
            dot.style.top = `${dotY}px`;

            requestAnimationFrame(animateDot);
        }

        animateDot(); // Start animation

        // Optional: Add hover effect on interactive elements
        const hoverTargets = document.querySelectorAll('button, a, lord-icon, i, svg, .hover-target');
        hoverTargets.forEach(el => {
            el.addEventListener('mouseenter', () => dot.classList.add('hovering'));
            el.addEventListener('mouseleave', () => dot.classList.remove('hovering'));
        });
    </script>
